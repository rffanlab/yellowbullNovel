## 一、系统概述

状态管理系统是 inkos 的核心基础设施，负责管理书籍的写锁、配置、控制文档、章节索引、状态快照、记忆数据库和运行时状态。Go 版本使用 GORM 实现多数据库支持（SQLite / MySQL / PostgreSQL）。

```
StateManager（状态管理入口）
  │
  ├─ 书籍写锁（acquireBookLock / releaseLock）
  ├─ 配置加载（loadBookConfig / saveBookConfig）
  ├─ 控制文档管理（ensureControlDocuments）
  ├─ 章节索引管理（loadChapterIndex / saveChapterIndex / getNextChapterNumber）
  ├─ 状态快照（snapshotState / restoreState）
  │
  ├─ MemoryDB（SQLite 记忆数据库）
  │   ├─ Fact 表（时序事实）
  │   ├─ StoredSummary 表（章节摘要）
  │   └─ StoredHook 表（伏笔记录）
  │
  ├─ RuntimeStateStore（运行时状态快照存储）
  │   ├─ loadRuntimeStateSnapshot
  │   ├─ buildRuntimeStateArtifacts
  │   ├─ saveRuntimeStateSnapshot
  │   └─ loadNarrativeMemorySeed
  │
  ├─ StateReducer（状态归约器）
  │   └─ applyRuntimeStateDelta（应用增量）
  │
  └─ StateBootstrap（从 Markdown 重建结构化状态）
      └─ bootstrapStructuredStateFromMarkdown
```

## 二、StateManager

### 2.1 书籍写锁

StateManager 实现了进程级书籍写锁，防止同一本书被多个进程同时写入。锁使用文件系统实现，支持心跳续约和过期自动回收。

#### 锁常量

```go
const (
    BookLockHeartbeatMs     = 30 * 1000     // 心跳间隔 30 秒
    BookLockLeaseMs         = 3 * 60 * 1000 // 租约有效期 3 分钟
    BookLockReleaseRetries  = 4             // 释放重试次数
)
```

#### 锁元数据

```go
type BookLockMetadata struct {
    Version     int    `json:"version"`     // 锁文件版本（固定 1）
    PID         int    `json:"pid"`         // 持有锁的进程 ID
    Token       string `json:"token"`       // UUID，唯一标识本次锁
    StartedAt   int64  `json:"startedAt"`   // 获取时间（毫秒时间戳）
    HeartbeatAt int64  `json:"heartbeatAt"` // 最近心跳时间（毫秒时间戳）
}
```

#### 获取锁流程

```
acquireBookLock(bookId)
  │
  ├─ 创建 bookDir 目录
  ├─ 构造 lockPath = bookDir/.write.lock
  ├─ 检查进程内锁表 processBookLocks
  │   └─ 已有同进程锁 → BookWriteLockError
  │
  ├─ 在 processBookLocks 中预留槽位（同步，防止同进程竞争）
  │
  ├─ 尝试创建锁文件（最多 4 次）
  │   ├─ createLockFile(lockPath, metadata) → 使用 O_WRONLY|O_CREAT|O_EXCL
  │   ├─ EEXIST（文件已存在）:
  │   │   ├─ 读取锁快照
  │   │   ├─ isStaleLock?
  │   │   │   ├─ 同 PID → 过期（当前进程的孤儿锁）
  │   │   │   ├─ PID 不存活 → 过期
  │   │   │   └─ heartbeat 超过 3 分钟 → 过期
  │   │   ├─ 过期 → removeStaleLock → 重试
  │   │   └─ 未过期 → BookWriteLockError
  │   └─ 其他错误 → 抛出
  │
  ├─ startLockHeartbeat(lockPath, lockKey, owner)
  │   └─ 每 30 秒更新 heartbeatAt
  │
  └─ 返回 releaseFunc
      └─ releaseFunc():
          ├─ 清除心跳定时器
          ├─ 等待心跳任务完成
          ├─ 从 processBookLocks 删除
          └─ unlinkWithRetry(lockPath)
```

#### BookWriteLockError

```go
type BookWriteLockError struct {
	BookID   string
	LockPath string
	LockData string
}

func (e *BookWriteLockError) Error() string {
	return fmt.Sprintf(`Book "%s" is locked by an active write (%s).
Wait for it to finish or stop the running task, then retry.
Stale locks are recovered automatically.`, e.BookID, e.LockData)
}
```

```go
func (sm *StateManager) EnsureControlDocumentsAt(bookDir string, language string, authorIntent *string) error {
	storyDir := filepath.Join(bookDir, "story")
	runtimeDir := filepath.Join(storyDir, "runtime")
	outlineDir := filepath.Join(storyDir, "outline")
	rolesMajorDir := filepath.Join(storyDir, "roles", "主要角色")
	rolesMinorDir := filepath.Join(storyDir, "roles", "次要角色")

	// 创建所有目录
	os.MkdirAll(storyDir, 0755)
	os.MkdirAll(runtimeDir, 0755)
	os.MkdirAll(outlineDir, 0755)
	os.MkdirAll(rolesMajorDir, 0755)
	os.MkdirAll(rolesMinorDir, 0755)

	// author_intent.md（如果缺失则创建默认值）
	authorIntentPath := filepath.Join(storyDir, "author_intent.md")
	if authorIntent != nil && strings.TrimSpace(*authorIntent) != "" {
		writeIfMissing(authorIntentPath, strings.TrimRight(*authorIntent, " \t\n\r")+"\n")
	} else {
		writeIfMissing(authorIntentPath, defaultAuthorIntent(language))
	}

	// current_focus.md（如果缺失则创建默认值)
	currentFocusPath := filepath.Join(storyDir, "current_focus.md")
	writeIfMissing(currentFocusPath, defaultCurrentFocus(language))

	// style_guide.md（确保包含写作方法论）
	styleGuidePath := filepath.Join(storyDir, "style_guide.md")
	ensureStyleGuide(styleGuidePath, language)

	return nil
}
```

### 2.4 章节索引管理

#### loadChapterIndex

```go
// LoadChapterIndex 加载章节索引
// 如果 index.json 不存在或为空，从章节文件重建
func (sm *StateManager) LoadChapterIndex(bookID string) ([]ChapterMeta, error) {
	indexPath := filepath.Join(sm.BookDir(bookID), "chapters", "index.json")
	raw, err := os.ReadFile(indexPath)
	if err == nil {
		var parsed []ChapterMeta
		if json.Unmarshal(raw, &parsed) == nil && len(parsed) > 0 {
			return parsed, nil
		}
		// 空数组 → 尝试从文件重建
		rebuilt := sm.rebuildChapterIndexFromFiles(bookID)
		if len(rebuilt) > 0 {
			return rebuilt, nil
		}
		return parsed, nil // 返回空数组
	}
	// 文件不存在 → 从文件重建
	rebuilt := sm.rebuildChapterIndexFromFiles(bookID)
	if len(rebuilt) > 0 {
		return rebuilt, nil
	}
	return []ChapterMeta{}, nil
}
```

#### rebuildChapterIndexFromFiles

从 `chapters/` 目录下的 `.md` 文件重建索引：

```go
func (sm *StateManager) rebuildChapterIndexFromFilesAt(bookDir string) []ChapterMeta {
	chaptersDir := filepath.Join(bookDir, "chapters")
	files, err := os.ReadDir(chaptersDir)
	if err != nil {
		return nil
	}

	var rows []ChapterMeta
	for _, file := range files {
		// 文件名格式: {number}_{title}.md 或 {number}-{title}.md
		match := regexp.MustCompile(`^(\d+)[_-]?(.*?)\.md$`).FindStringSubmatch(file.Name())
		if match == nil {
			continue
		}
		number, _ := strconv.Atoi(match[1])
		if number <= 0 {
			continue
		}
		filePath := filepath.Join(chaptersDir, file.Name())
		info, _ := os.Stat(filePath)
		content, _ := os.ReadFile(filePath)

		timestamp := time.Now().UTC().Format(time.RFC3339)
		if info != nil {
			timestamp = info.ModTime().UTC().Format(time.RFC3339)
		}
		rawTitle := strings.TrimSpace(
			strings.ReplaceAll(strings.TrimLeft(match[2], "_"), "_", " "),
		)

		rows = append(rows, ChapterMeta{
			Number:       number,
			Title:        rawTitle,
			Status:       "ready-for-review",
			WordCount:    len(strings.Fields(strings.ReplaceAll(string(content), " ", ""))),
			CreatedAt:    timestamp,
			UpdatedAt:    timestamp,
			AuditIssues:  []string{},
			LengthWarnings: []string{},
		})
	}

	sort.Slice(rows, func(i, j int) bool { return rows[i].Number < rows[j].Number })
	return rows
}
```

### 2.5 状态快照

`snapshotState` 在每章写入后将当前 truth files 复制到快照目录：

```go
// SnapshotStateAt 在指定目录创建状态快照
func (sm *StateManager) SnapshotStateAt(bookDir string, chapterNumber int) error {
	storyDir := filepath.Join(bookDir, "story")
	snapshotDir := filepath.Join(storyDir, "snapshots", strconv.Itoa(chapterNumber))
	os.MkdirAll(snapshotDir, 0755)

	// 复制 Markdown truth files
	files := []string{
		"current_state.md", "particle_ledger.md", "pending_hooks.md",
		"chapter_summaries.md", "subplot_board.md", "emotional_arcs.md",
		"character_matrix.md",
	}
	for _, f := range files {
		src := filepath.Join(storyDir, f)
		if content, err := os.ReadFile(src); err == nil {
			os.WriteFile(filepath.Join(snapshotDir, f), content, 0644)
		}
		// 文件不存在则跳过
	}

	// 复制结构化状态目录（state/*.json）
	stateDir := filepath.Join(bookDir, "story", "state")
	snapshotStateDir := filepath.Join(snapshotDir, "state")
	if entries, err := os.ReadDir(stateDir); err == nil && len(entries) > 0 {
		os.MkdirAll(snapshotStateDir, 0755)
		for _, entry := range entries {
			content, err := os.ReadFile(filepath.Join(stateDir, entry.Name()))
			if err == nil {
				os.WriteFile(filepath.Join(snapshotStateDir, entry.Name()), content, 0644)
			}
		}
	}

	return nil
}
```

### 2.6 状态恢复

`restoreState` 从快照恢复 truth files：

```go
// RestoreState 从快照恢复状态
func (sm *StateManager) RestoreState(bookID string, chapterNumber int) (bool, error) {
	storyDir := filepath.Join(sm.BookDir(bookID), "story")
	snapshotDir := filepath.Join(storyDir, "snapshots", strconv.Itoa(chapterNumber))

	requiredFiles := []string{"current_state.md", "pending_hooks.md"}
	optionalFiles := []string{
		"particle_ledger.md", "chapter_summaries.md",
		"subplot_board.md", "emotional_arcs.md", "character_matrix.md",
	}

	// 恢复必需文件
	for _, f := range requiredFiles {
		content, err := os.ReadFile(filepath.Join(snapshotDir, f))
		if err != nil {
			return false, err
		}
		os.WriteFile(filepath.Join(storyDir, f), content, 0644)
	}

	// 恢复可选文件（不存在则删除目标）
	for _, f := range optionalFiles {
		targetPath := filepath.Join(storyDir, f)
		content, err := os.ReadFile(filepath.Join(snapshotDir, f))
		if err == nil {
			os.WriteFile(targetPath, content, 0644)
		} else {
			os.Remove(targetPath)
		}
	}

	// 恢复结构化状态
	snapshotStateDir := filepath.Join(snapshotDir, "state")
	stateDir := sm.StateDir(bookID)
	if entries, err := os.ReadDir(snapshotStateDir); err == nil && len(entries) > 0 {
		os.MkdirAll(stateDir, 0755)
		for _, entry := range entries {
			content, _ := os.ReadFile(filepath.Join(snapshotStateDir, entry.Name()))
			os.WriteFile(filepath.Join(stateDir, entry.Name()), content, 0644)
		}
	}

	return true, nil
}
```

## 三、MemoryDB SQLite 记忆数据库

MemoryDB 是时序记忆数据库，存储带有时间效期的事实、章节摘要和伏笔记录。它是 Markdown truth files 的加速索引，支持精确查询如"第 5 章时角色 X 知道什么"。

### 3.1 数据库初始化

```go
type MemoryDB struct {
	db *gorm.DB
}

func NewMemoryDB(bookDir string) (*MemoryDB, error) {
	dbPath := filepath.Join(bookDir, "story", "memory.db")
	db, err := gorm.Open(sqlite.Open(dbPath), &gorm.Config{})
	if err != nil {
		return nil, err
	}

	// 启用 WAL 模式
	db.Exec("PRAGMA journal_mode = WAL")

	// 自动迁移
	if err := db.AutoMigrate(&Fact{}, &StoredSummary{}, &StoredHook{}); err != nil {
		return nil, err
	}

	// 创建索引
	db.Exec("CREATE INDEX IF NOT EXISTS idx_facts_subject ON facts(subject)")
	db.Exec("CREATE INDEX IF NOT EXISTS idx_facts_valid ON facts(valid_from_chapter, valid_until_chapter)")
	db.Exec("CREATE INDEX IF NOT EXISTS idx_facts_source ON facts(source_chapter)")
	db.Exec("CREATE INDEX IF NOT EXISTS idx_hooks_status ON hooks(status)")
	db.Exec("CREATE INDEX IF NOT EXISTS idx_hooks_last_advanced ON hooks(last_advanced_chapter)")

	return &MemoryDB{db: db}, nil
}
```

## 五、StateReducer 状态归约器

StateReducer 负责将 `RuntimeStateDelta`（增量）应用到当前快照，产出新快照。

### applyRuntimeStateDelta

```go
// ApplyRuntimeStateDelta 应用运行时状态增量
func ApplyRuntimeStateDelta(params ApplyRuntimeStateDeltaParams) (*RuntimeStateSnapshot, error)

type ApplyRuntimeStateDeltaParams struct {
	Snapshot      RuntimeStateSnapshot
	Delta         RuntimeStateDelta
	AllowReapply  bool // 允许重新应用同一章节（修复重试时为 true）
}
```

### 应用流程

```
applyRuntimeStateDelta(snapshot, delta, allowReapply)
  │
  ├─ 校验 delta.chapter 不回退
  │   ├─ allowReapply=true: delta.chapter < manifest.lastAppliedChapter → 报错
  │   └─ allowReapply=false: delta.chapter <= manifest.lastAppliedChapter → 报错
  │
  ├─ 校验 chapterSummary.chapter == delta.chapter
  │
  ├─ 校验不重复插入摘要（非 allowReapply 模式）
  │
  ├─ applyHookOps(snapshot.hooks, delta)
  │   ├─ 处理 delta.hookOps.upsert:
  │   │   ├─ 已存在 → mergeHookRecord（合并字段）
  │   │   ├─ 新增 → evaluateHookAdmission（准入检查）
  │   │   │   ├─ admit → 插入
  │   │   │   └─ duplicate_family → mergeDuplicateHookFamily
  │   │   └─ 排序（startChapter → lastAdvancedChapter → hookId）
  │   ├─ 处理 delta.hookOps.resolve:
  │   │   └─ 设置 status="resolved"
  │   └─ 处理 delta.hookOps.defer:
  │       └─ 设置 status="deferred"
  │
  ├─ applyCurrentStatePatch(snapshot.currentState, language, delta)
  │   ├─ 对每个 patch 字段（currentLocation/protagonistState/...）
  │   │   ├─ 删除旧的同谓语事实
  │   │   └─ 插入新事实
  │   └─ 排序事实（predicate → object）
  │
  ├─ applySummaryDelta(snapshot.chapterSummaries, delta, allowReapply)
  │   ├─ allowReapply → 删除同章节旧摘要 → 插入新摘要
  │   └─ 非 allowReapply → 直接追加
  │
  ├─ 更新 manifest.lastAppliedChapter = delta.chapter
  │
  ├─ validateRuntimeState(next) → 校验新快照
  │   └─ 有问题 → 抛错
  │
  └─ 返回新快照
```

### HookRecord 合并规则

```go
func mergeHookRecord(existing, incoming HookRecord) HookRecord {
	return HookRecord{
		// startChapter 取较小值
		StartChapter: min(existing.StartChapter, incoming.StartChapter),
		// type 取更丰富文本
		Type: preferRicherText(existing.Type, incoming.Type),
		// status 合并
		Status: mergeHookStatus(existing.Status, incoming.Status, progressed),
		// lastAdvancedChapter 取较大值
		LastAdvancedChapter: max(existing.LastAdvancedChapter, incoming.LastAdvancedChapter),
		// expectedPayoff 取更丰富文本
		ExpectedPayoff: preferRicherText(existing.ExpectedPayoff, incoming.ExpectedPayoff),
		// payoffTiming 解析
		PayoffTiming: resolveHookPayoffTiming(incoming.PayoffTiming, existing.PayoffTiming),
		// notes 取更丰富文本
		Notes: preferRicherText(existing.Notes, incoming.Notes),
	}
}

func mergeHookStatus(existing, incoming string, progressed bool) string {
	if existing == "resolved" || incoming == "resolved" {
		return "resolved"
	}
	if progressed || existing == "progressing" || incoming == "progressing" {
		return "progressing"
	}
	return existing
}
```

## 七、多数据库支持（GORM）

Go 版本使用 GORM 实现多数据库支持。通过不同的 Driver，同一套 Model 可以运行在 SQLite / MySQL / PostgreSQL 上。

### 数据库驱动

```go
// DatabaseConfig 数据库配置
type DatabaseConfig struct {
	Driver string // "sqlite" / "mysql" / "postgres"
	DSN    string // 数据源名称
}

// NewMemoryDBWithConfig 使用配置创建 MemoryDB
func NewMemoryDBWithConfig(bookDir string, dbConfig DatabaseConfig) (*MemoryDB, error) {
	var db *gorm.DB
	var err error

	switch dbConfig.Driver {
	case "sqlite":
		dbPath := filepath.Join(bookDir, "story", "memory.db")
		db, err = gorm.Open(sqlite.Open(dbPath), &gorm.Config{})
	case "mysql":
		db, err = gorm.Open(mysql.Open(dbConfig.DSN), &gorm.Config{})
	case "postgres":
		db, err = gorm.Open(postgres.Open(dbConfig.DSN), &gorm.Config{})
	default:
		return nil, fmt.Errorf("unsupported database driver: %s", dbConfig.Driver)
	}

	if err != nil {
		return nil, err
	}

	// SQLite 特定优化
	if dbConfig.Driver == "sqlite" {
		db.Exec("PRAGMA journal_mode = WAL")
	}

	// 自动迁移
	if err := db.AutoMigrate(&Fact{}, &StoredSummary{}, &StoredHook{}); err != nil {
		return nil, err
	}

	return &MemoryDB{db: db}, nil
}
```
