### 2.2 路由守卫

```typescript
// src/router/index.ts
import { createRouter, createWebHistory, type RouteRecordRaw } from 'vue-router'

const routes: RouteRecordRaw[] = [
  { path: '/', name: 'dashboard', component: () => import('@/views/Dashboard.vue') },
  { path: '/books/:id', name: 'book-detail', component: () => import('@/views/BookDetail.vue'), props: true },
  { path: '/books/:id/chapters/:num', name: 'chapter-reader', component: () => import('@/views/ChapterReader.vue'), props: true },
  { path: '/short-fiction', name: 'short-fiction', component: () => import('@/views/ShortFiction.vue') },
  { path: '/settings', name: 'settings', component: () => import('@/views/Settings.vue') },
  { path: '/:pathMatch(.*)*', name: 'not-found', component: () => import('@/views/NotFound.vue') },
]

const router = createRouter({
  history: createWebHistory(import.meta.env.BASE_URL),
  routes,
  scrollBehavior(_to, _from, saved) {
    return saved ?? { top: 0 }
  },
})

// 全局前置守卫：校验 bookId 存在性（可选）
router.beforeEach(async (to) => {
  if (to.name === 'book-detail' || to.name === 'chapter-reader') {
    const bookStore = useBookStore()
    const id = to.params.id as string
    if (!bookStore.current || bookStore.current.id !== id) {
      try {
        await bookStore.loadBook(id)
      } catch {
        return { name: 'dashboard' }
      }
    }
  }
})

export default router
```

### 2.3 页面职责

**Dashboard（`/`）**
- 书籍列表（卡片网格）：显示标题/题材/平台/当前章/状态/进度条。

- 左侧 `BookSidebar`：章节列表 / 角色 / 设定 / 伏笔 / 进度（tab 切换）。
- 右侧 `ChatPage`：与后端的对话主页，支持发"写下一章"等指令，SSE 流式渲染。

**ChapterReader（`/books/:id/chapters/:num`）**
- 章节正文 Markdown 渲染。
- 顶部：标题/字数/状态/审查分数。

**ShortFiction（`/short-fiction`）**
- 表单：标题/题材/平台/创意/目标字数。
- "开始生成" → SSE 流式渲染正文。

**Settings（`/settings`）**
- LLM 配置：提供商选择、API Key、Base URL、模型、温度。
- 项目设置：数据目录、调度默认参数。
- 测试连接按钮。

### 3.1 ChatPage（聊天主页）

**职责**：书籍的对话式操作入口，发送写作指令，SSE 流式渲染结果。

**Props**：`bookId: string`

**状态**：
- `messages: ChatMessage[]`：对话历史。
- `streaming: boolean`：是否正在流式接收。
- `currentStream: StreamState`：当前流式状态（阶段/chunk 累积）。

**交互**：
- 输入框 + 指令快捷按钮（"写下一章"/"审查第 N 章"/"润色第 N 章"）。
- 点击指令 → 调用 `useChatStore().startTask(...)` → 建 SSE 连接。
- 流式 chunk 实时追加到当前消息气泡。
- 阶段切换时显示进度条（planning 10% → writing 30% → reviewing 60% → polishing 90% → done 100%）。

**子组件**：
- `WriteProgress`：进度条 + 阶段标签 + token 用量。
- `ChatBubble`：单条消息气泡（支持 markdown 渲染）。
- `IssueList`：审查问题列表（折叠在消息内）。

## 四、状态管理 Pinia Stores

文件位置：`src/stores/`。

### 4.1 useBookStore（书籍状态）

```typescript
// src/stores/book.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { BookResponse, ChapterListItem, BookStatusResponse, TruthFilesResponse } from '@/api/types'
import * as bookApi from '@/api/book'

export const useBookStore = defineStore('book', () => {
  // === state ===
  const books = ref<BookResponse[]>([])
  const current = ref<BookResponse | null>(null)
  const chapters = ref<ChapterListItem[]>([])
  const status = ref<BookStatusResponse | null>(null)
  const truthFiles = ref<TruthFilesResponse | null>(null)
  const loading = ref(false)

  // === getters ===
  const activeBooks = computed(() => books.value.filter(b => b.status === 'active'))
  const currentChapter = computed(() => status.value?.currentChapter ?? 0)

  // === actions ===
  async function loadBooks() {
    loading.value = true
    try {
      books.value = await bookApi.listBooks()
    } finally {
      loading.value = false
    }
  }

  async function loadBook(id: string) {
    loading.value = true
    try {
      const [book, st, tf] = await Promise.all([
        bookApi.getBook(id),
        bookApi.getStatus(id),
        bookApi.getTruthFiles(id),
      ])
      current.value = book
      status.value = st
      truthFiles.value = tf
    } finally {
      loading.value = false
    }
  }

  async function createBook(req: bookApi.CreateBookRequest): Promise<BookResponse> {
    const book = await bookApi.createBook(req)
    books.value.push(book)
    return book
  }

  async function deleteBook(id: string) {
    await bookApi.deleteBook(id)
    books.value = books.value.filter(b => b.id !== id)
    if (current.value?.id === id) current.value = null
  }

  async function loadChapters(id: string) {
    chapters.value = await bookApi.listChapters(id)
  }

  async function refreshStatus() {
    if (!current.value) return
    status.value = await bookApi.getStatus(current.value.id)
  }

  return {
    books, current, chapters, status, truthFiles, loading,
    activeBooks, currentChapter,
    loadBooks, loadBook, createBook, deleteBook, loadChapters, refreshStatus,
  }
})
```

### 4.2 useChatStore（聊天状态）

```typescript
// src/stores/chat.ts
import { defineStore } from 'pinia'
import { ref, reactive } from 'vue'
import { createSSEClient, type SSEClient } from '@/api/sse'
import type { SSEEvent, StreamState } from '@/api/types'

export interface ChatMessage {
  id: string
  role: 'user' | 'assistant' | 'system'
  content: string
  timestamp: number
  // 流式附加
  stream?: StreamState
  issues?: AuditIssue[]
}

export const useChatStore = defineStore('chat', () => {
  const messages = ref<ChatMessage[]>([])
  const streaming = ref(false)
  const currentClient = ref<SSEClient | null>(null)

  function addMessage(msg: ChatMessage) {
    messages.value.push(msg)
  }

  function startTask(params: {
    bookId: string
    task: 'write-next' | 'review' | 'polish' | 'short-fiction'
    chapter?: number
    onChunk?: (chunk: string) => void
    onStage?: (stage: string, progress: number) => void
    onIssue?: (issue: AuditIssue) => void
    onDone?: (data: any) => void
    onError?: (err: Error) => void
  }) {
    if (streaming.value) return
    streaming.value = true

    const assistantMsg: ChatMessage = reactive({
      id: crypto.randomUUID(),
      role: 'assistant',
      content: '',
      timestamp: Date.now(),
      stream: { stage: '', stageLabel: '', progress: 0, issues: [], tokenUsage: null, streaming: true },
    })
    messages.value.push(assistantMsg)

    const client = createSSEClient({
      url: `/api/books/${params.bookId}/write-stream`,
      params: { task: params.task, chapter: params.chapter?.toString() ?? '' },
      onEvent: (ev: SSEEvent) => {
        switch (ev.type) {
          case 'stage':
            assistantMsg.stream!.stage = ev.stage
            assistantMsg.stream!.stageLabel = (ev.data as any)?.label ?? ev.stage
            assistantMsg.stream!.progress = Math.round(((ev.data as any)?.progress ?? 0) * 100)
            params.onStage?.(ev.stage, assistantMsg.stream!.progress)
            break
          case 'chunk':
            assistantMsg.content += (ev.data as any).content
            params.onChunk?.((ev.data as any).content)
            break
          case 'issue':
            const issue = (ev.data as any).issue
            assistantMsg.stream!.issues.push(issue)
            params.onIssue?.(issue)
            break
          case 'tokenUsage':
            assistantMsg.stream!.tokenUsage = ev.data
            break
          case 'done':
            assistantMsg.stream!.streaming = false
            assistantMsg.stream!.progress = 100
            params.onDone?.(ev.data)
            streaming.value = false
            client.close()
            break
          case 'error':
            assistantMsg.stream!.streaming = false
            params.onError?.(new Error((ev.data as any)?.message ?? 'unknown'))
            streaming.value = false
            client.close()
            break
        }
      },
      onError: (err: Error) => {
        assistantMsg.stream!.streaming = false
        params.onError?.(err)
        streaming.value = false
      },
    })
    client.start()
    currentClient.value = client
  }

  function cancelTask() {
    currentClient.value?.close()
    streaming.value = false
    const last = messages.value[messages.value.length - 1]
    if (last?.stream) last.stream.streaming = false
  }

  function clearMessages() {
    messages.value = []
  }

  return { messages, streaming, currentClient, addMessage, startTask, cancelTask, clearMessages }
})
```

### 5.1 axios 实例封装

```typescript
// src/api/client.ts
import axios, { type AxiosInstance, type AxiosRequestConfig } from 'axios'
import { ElMessage } from 'element-plus'

const client: AxiosInstance = axios.create({
  baseURL: import.meta.env.VITE_API_BASE ?? '/api',
  timeout: 300_000, // 长任务最长 5 分钟
  headers: { 'Content-Type': 'application/json' },
})

// 请求拦截器
client.interceptors.request.use((config) => {
  const token = localStorage.getItem('auth_token')
  if (token) config.headers.Authorization = `Bearer ${token}`
  return config
})

// 响应拦截器：统一解包 ApiResponse
client.interceptors.response.use(
  (resp) => {
    const body = resp.data
    if (body && typeof body === 'object' && 'code' in body) {
      if (body.code !== 0) {
        ElMessage.error(body.message ?? '请求失败')
        return Promise.reject(new Error(body.message))
      }
      return body.data // 解包
    }
    return body
  },
  (err) => {
    if (err.response?.status === 401) {
      localStorage.removeItem('auth_token')
      window.location.href = '/settings'
    }
    const msg = err.response?.data?.message ?? err.message
    ElMessage.error(msg)
    return Promise.reject(err)
  },
)

export { client }
```

## 六、SSE 客户端实现

文件位置：`src/api/sse.ts`。

### 6.1 设计要点

- 使用浏览器原生 `EventSource` API，不引入额外依赖。
- 支持 GET 请求（EventSource 限制）；复杂参数走 query string。
- 支持自定义事件类型（`event: stage` / `event: chunk` / ...）。
- 自动重连（EventSource 内置，但限制最大重连次数避免死循环）。
- 支持手动取消（`close()`）。
- 后端心跳（`ping` 注释行）由 EventSource 自动忽略。

### 6.2 实现

```typescript
// src/api/sse.ts
import type { SSEEvent } from './types'

export interface SSEClientOptions {
  url: string
  params?: Record<string, string>
  onEvent?: (ev: SSEEvent) => void
  onError?: (err: Error) => void
  onOpen?: () => void
  maxRetries?: number // 默认 3
}

export interface SSEClient {
  start(): void
  close(): void
  readonly readyState: number // 0=connecting, 1=open, 2=closed
}

export function createSSEClient(opts: SSEClientOptions): SSEClient {
  let es: EventSource | null = null
  let retries = 0
  let closed = false
  const maxRetries = opts.maxRetries ?? 3

  function buildURL(): string {
    const base = opts.url
    if (!opts.params) return base
    const qs = new URLSearchParams(
      Object.fromEntries(Object.entries(opts.params).filter(([, v]) => v !== '')),
    ).toString()
    return qs ? `${base}?${qs}` : base
  }

  function start() {
    if (closed) return
    const url = buildURL()
    es = new EventSource(url)

    es.onopen = () => {
      retries = 0
      opts.onOpen?.()
    }

    // 监听所有命名事件
    const eventTypes = ['stage', 'progress', 'chunk', 'tokenUsage', 'issue', 'note', 'error', 'done']
    for (const t of eventTypes) {
      es.addEventListener(t, (raw: MessageEvent) => {
        try {
          const ev = JSON.parse(raw.data) as SSEEvent
          opts.onEvent?.(ev)
        } catch (e) {
          // 忽略解析失败的事件（如心跳）
        }
      })
    }

    // 默认 message 事件（未命名的 data: 行）
    es.onmessage = (raw: MessageEvent) => {
      try {
        const ev = JSON.parse(raw.data) as SSEEvent
        opts.onEvent?.(ev)
      } catch {
        // 忽略
      }
    }

    es.onerror = () => {
      if (closed) return
      es?.close()
      if (retries < maxRetries && es?.readyState === EventSource.CLOSED) {
        retries++
        // 指数退避
        setTimeout(() => start(), 1000 * Math.pow(2, retries - 1))
      } else {
        opts.onError?.(new Error('SSE connection failed after retries'))
      }
    }
  }

  function close() {
    closed = true
    es?.close()
    es = null
  }

  return {
    start,
    close,
    get readyState() { return es?.readyState ?? 0 },
  }
}
```

### 6.3 使用示例

```typescript
// 在 ChatPage 中触发写作流式
const client = createSSEClient({
  url: `/api/books/${bookId}/write-stream`,
  params: { task: 'write-next' },
  onEvent: (ev) => {
    if (ev.type === 'chunk') {
      currentContent.value += ev.data.content
    } else if (ev.type === 'done') {
      ElMessage.success(`第 ${ev.data.chapterNumber} 章完成，${ev.data.wordCount} 字`)
    }
  },
  onError: (err) => ElMessage.error(err.message),
})
client.start()

// 取消
client.close()
```

### 6.4 注意事项

- **EventSource 仅支持 GET**：所有 SSE 端点必须是 GET，参数走 query。
- **无自定义请求头**：EventSource 不支持设置 Authorization 头。认证通过 cookie 或 query token。开发模式可关闭认证。
- **浏览器并发限制**：同源 EventSource 默认上限 6 个连接。单页同时只开 1 个 SSE 即可。
- **代理缓冲**：后端需设置 `X-Accel-Buffering: no`（nginx）避免事件被缓冲。
- **心跳**：后端每 15s 发一行 `: ping` 注释，EventSource 自动忽略，但保持连接活跃。
