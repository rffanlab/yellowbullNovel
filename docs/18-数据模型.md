## 一、书籍配置 BookConfig

书籍级配置，对应 `books/<book-id>/book.json`。参考 inkos `models/book.ts` 的 `BookConfigSchema`。

```go
// Platform 目标发布平台
type Platform string

const (
    PlatformTomato  Platform = "tomato"   // 番茄小说
    PlatformFeilu   Platform = "feilu"    // 飞卢
    PlatformQidian  Platform = "qidian"   // 起点
    PlatformOther   Platform = "other"    // 其他
)

// BookStatus 书籍生命周期状态
type BookStatus string

const (
    BookStatusIncubating BookStatus = "incubating"  // 孵化中（未建骨架）
    BookStatusOutlining  BookStatus = "outlining"   // 大纲生成中
    BookStatusActive     BookStatus = "active"      // 正常写作中
    BookStatusPaused     BookStatus = "paused"      // 暂停
    BookStatusCompleted  BookStatus = "completed"   // 完结
    BookStatusDropped    BookStatus = "dropped"     // 弃坑
)

// FanficMode 同人模式
type FanficMode string

const (
    FanficModeCanon FanficMode = "canon"  // 正典（遵循原作）
    FanficModeAU    FanficMode = "au"     // 平行宇宙
    FanficModeOOC   FanficMode = "ooc"    // 性格偏离
    FanficModeCP    FanficMode = "cp"     // 配对/感情线
)

// ChapterReviewMode 章节审查模式
type ChapterReviewMode string

const (
    ChapterReviewModeAuto   ChapterReviewMode = "auto"   // 自动审查（默认）
    ChapterReviewModeManual ChapterReviewMode = "manual" // 手动审查（写完即停）
)

// RevisionGate 修订门策略
type RevisionGate string

const (
    RevisionGateStrict  RevisionGate = "strict"  // 严格：不恶化且至少一项改善才应用
    RevisionGateLenient RevisionGate = "lenient" // 宽松：不恶化即应用
    RevisionGateAlways  RevisionGate = "always"  // 总是应用
)

// Language 语言
type Language string

const (
    LanguageZh Language = "zh"
    LanguageEn Language = "en"
)

// BookWritingConfig 书籍级写作配置（覆盖项目级）
type BookWritingConfig struct {
    // ReviewMode 章节审查模式，覆盖项目级 writing.reviewMode
    ReviewMode ChapterReviewMode `json:"reviewMode,omitempty" gorm:"column:review_mode;type:varchar(16)"`
    // RevisionGate 手动修订门策略，覆盖项目级 writing.revisionGate
    RevisionGate RevisionGate `json:"revisionGate,omitempty" gorm:"column:revision_gate;type:varchar(16)"`
}

// BookConfig 书籍配置
type BookConfig struct {
    // ID 书籍唯一标识（用于目录名 books/<id>/）
    ID string `json:"id" gorm:"column:id;primaryKey;type:varchar(128)"`

    // Title 书名
    Title string `json:"title" gorm:"column:title;type:varchar(512);not null"`

    // Platform 目标发布平台
    Platform Platform `json:"platform" gorm:"column:platform;type:varchar(32);index"`

    // Genre 题材标识（对应 genres/*.md 的 id，如 cultivation/xianxia/urban）
    Genre string `json:"genre" gorm:"column:genre;type:varchar(64);index"`

    // Status 书籍当前生命周期状态
    Status BookStatus `json:"status" gorm:"column:status;type:varchar(32);index"`

    // TargetChapters 目标章节数（默认 200）
    TargetChapters int `json:"targetChapters" gorm:"column:target_chapters;type:int;default:200"`

    // ChapterWordCount 每章目标字数（默认 3000，最小 1000）
    ChapterWordCount int `json:"chapterWordCount" gorm:"column:chapter_word_count;type:int;default:3000"`

    // Language 写作语言 zh|en
    Language Language `json:"language,omitempty" gorm:"column:language;type:varchar(8)"`

    // CreatedAt 创建时间（RFC3339）
    CreatedAt string `json:"createdAt" gorm:"column:created_at;type:datetime"`

    // UpdatedAt 最后更新时间
    UpdatedAt string `json:"updatedAt" gorm:"column:updated_at;type:datetime"`

    // ParentBookId 父书 ID（同人/番外/系列续写时指向正传）
    ParentBookID string `json:"parentBookId,omitempty" gorm:"column:parent_book_id;type:varchar(128);index"`

    // FanficMode 同人模式（仅同人书有值）
    FanficMode FanficMode `json:"fanficMode,omitempty" gorm:"column:fanfic_mode;type:varchar(16)"`

    // Writing 书籍级写作配置（覆盖项目级）
    Writing *BookWritingConfig `json:"writing,omitempty" gorm:"column:writing;type:json"`
}

// TableName 指定 GORM 表名
func (BookConfig) TableName() string { return "books" }
```

**字段用途说明**：

| 字段 | 用途 |
|------|------|
| `ID` | 书籍唯一标识，同时是 `books/<id>/` 目录名 |
| `Title` | 书名，显示和章节标题生成时使用 |
| `Platform` | 目标平台，影响字数统计规则和审查维度 |
| `Genre` | 题材标识，对应 `genres/*.md` 题材配置文件 |
| `Status` | 生命周期状态，调度器据此决定是否自动写作 |
| `TargetChapters` | 目标章节数，规划阶段用于推导卷级 OKR |
| `ChapterWordCount` | 每章目标字数，生成 LengthSpec 的基准 |
| `Language` | 写作语言，影响黄金开篇纪律（中文前3章/英文前5章） |
| `ParentBookID` | 同人/番外/系列续写时指向正传书 ID |
| `FanficMode` | 同人模式，影响审查维度（正典事件一致性等） |
| `Writing.ReviewMode` | auto=写完自动审查循环；manual=写完即停等用户手动触发 |
| `Writing.RevisionGate` | 控制手动修订的应用条件 |

## 二、章节元数据 ChapterMeta

章节元数据，对应 `books/<book-id>/chapter-index.json` 中的每一项。参考 inkos `models/chapter.ts` 的 `ChapterMetaSchema`。

```go
// ChapterMeta 章节元数据
type ChapterMeta struct {
    // Number 章节序号（从 1 开始）
    Number int `json:"number" gorm:"column:number;primaryKey;type:int"`

    // Title 章节标题
    Title string `json:"title" gorm:"column:title;type:varchar(512)"`

    // Status 章节当前状态
    Status ChapterStatus `json:"status" gorm:"column:status;type:varchar(32);index"`

    // WordCount 章节字数（最终字数，经过归一化后）
    WordCount int `json:"wordCount" gorm:"column:word_count;type:int;default:0"`

    // CreatedAt 创建时间
    CreatedAt string `json:"createdAt" gorm:"column:created_at;type:datetime"`

    // UpdatedAt 最后更新时间
    UpdatedAt string `json:"updatedAt" gorm:"column:updated_at;type:datetime"`

    // AuditIssues 审计问题摘要列表（人类可读）
    AuditIssues []string `json:"auditIssues" gorm:"column:audit_issues;type:json"`

    // LengthWarnings 字数警告列表
    LengthWarnings []string `json:"lengthWarnings" gorm:"column:length_warnings;type:json"`

    // ReviewNote 审查备注（状态降级原因等）
    ReviewNote string `json:"reviewNote,omitempty" gorm:"column:review_note;type:text"`

    // DetectionScore AIGC 检测分数（0-1，越高越像 AI 生成）
    DetectionScore *float64 `json:"detectionScore,omitempty" gorm:"column:detection_score;type:decimal(4,3)"`

    // DetectionProvider 检测服务提供商标识
    DetectionProvider string `json:"detectionProvider,omitempty" gorm:"column:detection_provider;type:varchar(64)"`

    // DetectedAt 检测时间
    DetectedAt string `json:"detectedAt,omitempty" gorm:"column:detected_at;type:datetime"`

    // LengthTelemetry 字数遥测（详细记录各阶段字数）
    LengthTelemetry *LengthTelemetry `json:"lengthTelemetry,omitempty" gorm:"column:length_telemetry;type:json"`

    // TokenUsage Token 消耗统计
    TokenUsage *TokenUsage `json:"tokenUsage,omitempty" gorm:"column:token_usage;type:json"`
}

// TableName 指定 GORM 表名
func (ChapterMeta) TableName() string { return "chapters" }
```

## 三、章节状态 ChapterStatus

章节在写作流水线中的状态机。参考 inkos `models/chapter.ts` 的 `ChapterStatusSchema`。

```go
// ChapterStatus 章节状态枚举
type ChapterStatus string

const (
    // ChapterStatusCardGenerated 章节卡片已生成（规划阶段产出 Intent）
    ChapterStatusCardGenerated ChapterStatus = "card-generated"
    // ChapterStatusDrafting 写作中
    ChapterStatusDrafting ChapterStatus = "drafting"
    // ChapterStatusDrafted 初稿完成
    ChapterStatusDrafted ChapterStatus = "drafted"
    // ChapterStatusAuditing 审计中
    ChapterStatusAuditing ChapterStatus = "auditing"
    // ChapterStatusAuditPassed 审计通过
    ChapterStatusAuditPassed ChapterStatus = "audit-passed"
    // ChapterStatusAuditFailed 审计未通过
    ChapterStatusAuditFailed ChapterStatus = "audit-failed"
    // ChapterStatusStateDegraded 状态降级（真相文件校验失败）
    ChapterStatusStateDegraded ChapterStatus = "state-degraded"
    // ChapterStatusRevising 修订中
    ChapterStatusRevising ChapterStatus = "revising"
    // ChapterStatusReadyForReview 就绪待审查（自动审查完成，等用户确认）
    ChapterStatusReadyForReview ChapterStatus = "ready-for-review"
    // ChapterStatusApproved 用户已通过
    ChapterStatusApproved ChapterStatus = "approved"
    // ChapterStatusRejected 用户已拒绝
    ChapterStatusRejected ChapterStatus = "rejected"
    // ChapterStatusPublished 已发布
    ChapterStatusPublished ChapterStatus = "published"
    // ChapterStatusImported 导入（导入续写场景）
    ChapterStatusImported ChapterStatus = "imported"
)
```

**状态流转**：

```
card-generated → drafting → drafted → auditing
                                       ├─→ audit-passed → ready-for-review → approved → published
                                       ├─→ audit-failed → revising → (回 auditing)
                                       └─→ state-degraded → (修复后回 ready-for-review)

imported (导入续写，跳过写作阶段)
```

## 四、章节意图 ChapterIntent

确定性章节意图，由 Planner Agent 产出（非 LLM 生成部分）。对应 `runtime/chapter-XXXX.intent.md` 的结构化部分。参考 inkos `models/input-governance.ts` 的 `ChapterIntentSchema`。

```go
// ChapterIntent 确定性章节意图
// 由 Planner 基于 大纲/当前状态/作者意图 推导，不依赖 LLM 生成
type ChapterIntent struct {
    // Chapter 章节序号
    Chapter int `json:"chapter" gorm:"column:chapter;type:int"`

    // Goal 章节目标（一句话，优先级链：外部指令→局部覆盖→大纲节点→当前聚焦→作者意图→默认）
    Goal string `json:"goal" gorm:"column:goal;type:varchar(256)"`

    // OutlineNode 当前章在大纲中的位置标识（卷号/章节范围/锚点）
    OutlineNode string `json:"outlineNode,omitempty" gorm:"column:outline_node;type:varchar(256)"`

    // ArcContext 弧线上下文（当前处于哪个叙事弧线的哪个阶段）
    ArcContext string `json:"arcContext,omitempty" gorm:"column:arc_context;type:text"`

    // MustKeep 必须保留的要素（写手不能违背的硬约束）
    MustKeep []string `json:"mustKeep" gorm:"column:must_keep;type:json"`

    // MustAvoid 必须避免的要素（写手不能触碰的禁忌）
    MustAvoid []string `json:"mustAvoid" gorm:"column:must_avoid;type:json"`

    // StyleEmphasis 文风强调点（如"紧凑节奏""冷峻叙事"等）
    StyleEmphasis []string `json:"styleEmphasis" gorm:"column:style_emphasis;type:json"`
}
```

**字段用途说明**：

| 字段 | 用途 |
|------|------|
| `Goal` | 章节的核心目标，一句话描述。Planner 通过优先级链推导，是写作方向的权威依据 |
| `OutlineNode` | 当前章在 volume_map 中的定位，用于语义选段时选择相关大纲段落 |
| `ArcContext` | 叙事弧线上下文，如"第一卷·觉醒篇·中段"，帮助 Writer 把握节奏 |
| `MustKeep` | 硬约束列表，如"必须推进主角觉醒""必须回收第3章伏笔" |
| `MustAvoid` | 禁忌列表，如"不能让配角知道主角身份""不能出现现代科技" |
| `StyleEmphasis` | 文风指导，如"冷峻""快节奏""多用对话推进" |

## 五、章节备忘 ChapterMemo

LLM 生成的 7 段章节备忘，由 Planner Agent 调用 LLM 产出。是方向性指导，不写具体情节，留给 Writer 发挥空间。参考 inkos `models/input-governance.ts` 的 `ChapterMemoSchema`。

```go
// ChapterMemo LLM 生成的章节备忘
// 由 Planner 调用 LLM 生成，包含 7 段方向性指导
type ChapterMemo struct {
    // Chapter 章节序号
    Chapter int `json:"chapter" gorm:"column:chapter;type:int"`

    // Goal 本章目标（≤50字，一句话概括）
    Goal string `json:"goal" gorm:"column:goal;type:varchar(128)"`

    // IsGoldenOpening 是否为黄金开篇章节（中文前3章/英文前5章为 true）
    // 黄金开篇章节有特殊的写作纪律指导
    IsGoldenOpening bool `json:"isGoldenOpening" gorm:"column:is_golden_opening;type:boolean;default:false"`

    // ThreadRefs 关联线索 ID 列表（伏笔/支线/角色弧线引用）
    ThreadRefs []string `json:"threadRefs" gorm:"column:thread_refs;type:json"`

    // Body 备忘正文（7 段 Markdown 文本）
    // 包含以下 7 个段落：
    //   1. 本章目标 — 本章要达成的叙事目标
    //   2. 关联线索 — 本章关联的伏笔/支线/角色弧线
    //   3. 当前任务 — 本章推进的具体任务
    //   4. 读者在等什么 — 读者期待兑现的内容
    //   5. 该兑现的/暂不掀的 — 哪些伏笔该兑现，哪些暂不揭底
    //   6. 日常过渡承担什么 — 如果是过渡章，承担什么叙事功能
    //   7. 关键抉择三连问 — 本章关键抉择的三连问
    //   8. 章尾必须发生的改变 — 章尾必须达成的状态改变
    //   9. 本章 hook 账 — 本章要埋/推进/回收的伏笔清单
    //  10. 不要做 — 本章明确不要做的事
    Body string `json:"body" gorm:"column:body;type:text"`
}
```

**7 段备忘设计要点**：

| 段落 | 作用 |
|------|------|
| 本章目标 | 方向性目标，不写具体情节 |
| 关联线索 | 引用伏笔 ID，Composer 据此收集伏笔债务 |
| 当前任务 | 具体要推进的叙事任务 |
| 读者在等什么 | 防止跳票，确保兑现读者期待 |
| 该兑现的/暂不掀的 | 伏笔预算控制，决定哪些掀底哪些保留 |
| 日常过渡承担什么 | 过渡章也要有叙事功能，不能纯水 |
| 关键抉择三连问 | 强制 Writer 思考角色的关键抉择 |
| 章尾必须发生的改变 | 确保"章尾有钩子"，避免平淡收尾 |
| 本章 hook 账 | 伏笔记账，与 pending_hooks.md 对账 |
| 不要做 | 负面清单，防止 Writer 跑偏 |

## 六、上下文包 ContextPackage

Composer Agent 组装的上下文包，是 Writer 的输入。对应 `runtime/chapter-XXXX.context.json`。参考 inkos `models/input-governance.ts` 的 `ContextPackageSchema`。

```go
// ContextSource 上下文来源条目
type ContextSource struct {
    // Source 来源标识（如 "story_frame"/"current_state"/"chapter_memo"/"memory:fact" 等）
    Source string `json:"source" gorm:"column:source;type:varchar(128)"`

    // Reason 为什么选择此上下文（供 trace 追踪和调试）
    Reason string `json:"reason" gorm:"column:reason;type:text"`

    // Excerpt 上下文摘要文本（实际喂给 Writer 的内容）
    Excerpt string `json:"excerpt,omitempty" gorm:"column:excerpt;type:text"`
}

// ContextPackage 上下文包
type ContextPackage struct {
    // Chapter 章节序号
    Chapter int `json:"chapter" gorm:"column:chapter;type:int"`

    // SelectedContext 选中的上下文条目列表
    // 包含来源：chapter_memo / current_focus / author_intent / audit_drift /
    //          current_state / story_frame / volume_map / parent_canon /
    //          近5章摘要 trail / 伏笔债务 / 记忆检索 / 角色矩阵 等
    SelectedContext []ContextSource `json:"selectedContext" gorm:"column:selected_context;type:json"`
}
```

**上下文来源分类**：

| 类别 | 来源 | protected? |
|------|------|-----------|
| 方向指导 | `chapter_memo`, `current_focus`, `author_intent` | ✅ protected |
| 硬状态 | `current_state`, `audit_drift` | ✅ protected |
| 大纲选段 | `story_frame`, `volume_map` | ❌ compressible |
| 同人正典 | `parent_canon`, `fanfic_canon` | ❌ compressible |
| 近章摘要 | `chapter_summaries`（近5章 trail） | ❌ compressible |
| 伏笔债务 | `hooks`（memo 引用的原始种子） | ✅ protected |
| 记忆检索 | `memory:fact`, `memory:summary`, `memory:hook` | ❌ compressible |
| 角色关系 | `character_matrix`, `emotional_arcs` | ❌ compressible |

**protected vs compressible**：超上下文预算时，protected 内容绝不压缩，compressible 内容由 LLM 语义编译压缩。

## 七、治理规则栈 RuleStack

治理规则栈，从 Intent 提取的硬/软规则分层。对应 `runtime/chapter-XXXX.rulestack.json`。参考 inkos `models/input-governance.ts` 的 `RuleStackSchema`。

```go
// RuleLayerScope 规则层作用域
type RuleLayerScope string

const (
    RuleLayerScopeGlobal RuleLayerScope = "global" // 全局规则（适用于所有书）
    RuleLayerScopeBook   RuleLayerScope = "book"   // 书级规则（单本书）
    RuleLayerScopeArc    RuleLayerScope = "arc"    // 弧线级规则（当前叙事弧线）
    RuleLayerScopeLocal  RuleLayerScope = "local"  // 局部规则（单章）
)

// RuleLayer 规则层
type RuleLayer struct {
    // ID 层唯一标识
    ID string `json:"id" gorm:"column:id;type:varchar(128)"`

    // Name 层名称（人类可读）
    Name string `json:"name" gorm:"column:name;type:varchar(256)"`

    // Precedence 优先级（数字越大优先级越高，高优先级覆盖低优先级）
    Precedence int `json:"precedence" gorm:"column:precedence;type:int"`

    // Scope 作用域
    Scope RuleLayerScope `json:"scope" gorm:"column:scope;type:varchar(32)"`
}

// OverrideEdge 覆盖边（定义哪些层可以覆盖哪些层）
type OverrideEdge struct {
    // From 覆盖方层 ID
    From string `json:"from" gorm:"column:from;type:varchar(128)"`

    // To 被覆盖方层 ID
    To string `json:"to" gorm:"column:to;type:varchar(128)"`

    // Allowed 是否允许覆盖
    Allowed bool `json:"allowed" gorm:"column:allowed;type:boolean"`

    // Scope 作用域标识
    Scope string `json:"scope" gorm:"column:scope;type:varchar(64)"`
}

// ActiveOverride 活跃覆盖（当前生效的覆盖关系）
type ActiveOverride struct {
    // From 覆盖方层 ID
    From string `json:"from" gorm:"column:from;type:varchar(128)"`

    // To 被覆盖方层 ID
```
